#!/usr/bin/env python
class Person:
	def __init__(self, name, age, university, course, year, hobbies):
		self.name = name
		self.age = age
		self.university = university
		self.course = course
		self.year = year
		self.hobbies = hobbies

	def myfunc(self):
		print("Hello! My name is " + self.name)
		print("I am " + self.age + " years old")
		print("I am studying " + self.course + " at " + self.university)
		print("I am currently in " + self.year)
		print("My hobbies are " + self.hobbies)

introduction = Person("Liam Eguia", "20", "Dublin City University", "Computer Science", "3rd Year", "Tennis, Gym, Reading, Socialising, Travelling")

introduction.myfunc()
